<objective>
Replace the manual fixed-array expected-token accumulation in the PEG engine with `std.EnumSet(TokenKind)`.

Purpose: The current implementation uses two `[64]TokenKind` buffers with manual dedup-on-read. `EnumSet` provides O(1) insert with automatic deduplication, eliminates buffer overflow risk, and removes ~20 lines of manual dedup code. This is a cleanup of the Phase 31 implementation to use the idiomatic Zig data structure.

Output: Cleaner engine.zig with EnumSet, updated consumers in module.zig and commands.zig.
</objective>

3. **trackFailure** (lines 226-243): Simplify to:

   ```zig
   fn trackFailure(self: *Engine, pos: usize, expected: TokenKind) void {
       if (pos > self.furthest_pos) {
           // New furthest position — reset set
           self.furthest_pos = pos;
           self.furthest_expected = std.EnumSet(TokenKind).initEmpty();
           self.furthest_expected.insert(expected);
       } else if (pos == self.furthest_pos) {
           // Same position — accumulate
           self.furthest_expected.insert(expected);
       }
       // pos < furthest_pos: ignore
   }
   ```

   ```zig
   pub fn getError(self: *Engine) ParseError {
       const pos = @min(self.furthest_pos, if (self.tokens.len > 0) self.tokens.len - 1 else 0);
       const tok = if (pos < self.tokens.len) self.tokens[pos] else Token{
           .kind = .eof, .text = "", .line = 0, .col = 0,
       };
       return .{
           .pos = pos,
           .line = tok.line,
           .col = tok.col,
           .found = tok.text,
   ```

Rewrite the body to use `.count()` and `.iterator()`:
```zig
fn formatExpectedSet(alloc: std.mem.Allocator, set: std.EnumSet(lexer.TokenKind)) ![]u8 {
    const engine_mod = @import("peg/engine.zig");
    const total = set.count();
    var buf = std.ArrayListUnmanaged(u8){};
    errdefer buf.deinit(alloc);
    try buf.appendSlice(alloc, "expected ");
    var it = set.iterator();
    var i: usize = 0;
    while (it.next()) |kind| {
        if (i > 0 and i < total - 1) try buf.appendSlice(alloc, ", ");
        if (i > 0 and i == total - 1) {
            if (total > 2) try buf.appendSlice(alloc, ", or ") else try buf.appendSlice(alloc, " or ");
        }
        try buf.append(alloc, '\'');
        try buf.appendSlice(alloc, engine_mod.kindDisplayName(kind));
        try buf.append(alloc, '\'');
        i += 1;
    }
    return buf.toOwnedSlice(alloc);
}
```

Also update the call site (line 475): change `.len > 1` to `.count() > 1`.
