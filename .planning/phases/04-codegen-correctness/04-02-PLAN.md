fn annotateCallCoercions(self: *MirAnnotator, c: parser.CallExpr) !void {
    const sig = self.resolveCallSig(c) orelse return;
    const param_count = @min(c.args.len, sig.params.len);
    for (c.args[0..param_count], sig.params[0..param_count]) |arg, param| {
        const arg_type = self.lookupType(arg) orelse continue;
        const coercion = detectCoercion(arg_type, param.type_);
        if (coercion.kind) |kind| {
            try self.node_map.put(self.allocator, arg, .{
                .resolved_type = arg_type,
                .type_class = classifyType(arg_type),
                .coercion = kind,
                .coerce_tag = coercion.tag,
            });
        }
    }
}

fn resolveCallSig(self: *const MirAnnotator, c: parser.CallExpr) ?declarations.FuncSig {
    // Direct call: func_name(args)
    if (c.callee.* == .identifier) {
        return self.decls.funcs.get(c.callee.identifier);
    }
    // Module call: module.func_name(args) or module.Type.method(args)
    if (c.callee.* == .field_expr) {
        const local = self.decls.funcs.get(c.callee.field_expr.field);
        if (local != null) return local;
        // Cross-module: module.func_name
        if (c.callee.field_expr.object.* == .identifier) {
            if (self.all_decls) |ad| {
                if (ad.get(c.callee.field_expr.object.identifier)) |mod_decls| {
                    return mod_decls.funcs.get(c.callee.field_expr.field);
                }
            }
        }
        // Cross-module: module.Type.method
        if (c.callee.field_expr.object.* == .field_expr) {
            const inner = c.callee.field_expr.object.field_expr;
            if (inner.object.* == .identifier) {
                if (self.all_decls) |ad| {
                    if (ad.get(inner.object.identifier)) |mod_decls| {
                        return mod_decls.funcs.get(c.callee.field_expr.field);
                    }
                }
            }
        }
    }
    return null;
}

**CGEN-02 fix in src/mir.zig — single approach: type-guided instance method resolution.**

The problem: `resolveCallSig` returns `null` for instance method calls like `obj.method(args)` where `obj` is a variable of a cross-module struct type. In this pattern, `c.callee.field_expr.object` is the variable name (e.g., `counter`), not a module name, so the `ad.get()` lookup fails.

The fix: In `resolveCallSig`, add a final fallback BEFORE `return null` (after line 585) that uses `self.lookupType()` on the callee's object node to resolve the variable's type, then searches the correct module's DeclTable for the method. This is type-guided, not a blind iterator.

Add this block:

```zig
// Instance method: obj.method(args) — resolve obj's type to find the declaring module
if (c.callee.* == .field_expr) {
    const obj_node = c.callee.field_expr.object;
    const method_name = c.callee.field_expr.field;
    if (self.lookupType(obj_node)) |obj_type| {
        // Extract struct name from resolved type to find the declaring module
        const struct_name = switch (obj_type) {
            .struct_type => |s| s.name,
            else => null,
        };
        if (struct_name) |sname| {
            // If struct name is qualified (e.g., "math.Counter"), extract module name
            if (std.mem.indexOf(u8, sname, ".")) |dot| {
                const mod_name = sname[0..dot];
                if (self.all_decls) |ad| {
                    if (ad.get(mod_name)) |mod_decls| {
                        if (mod_decls.funcs.get(method_name)) |fsig| {
                            return fsig;
                        }
                    }
                }
            }
        }
        // Fallback: struct name not qualified — search all modules for the method
        if (self.all_decls) |ad| {
            var it = ad.iterator();
            while (it.next()) |entry| {
                if (entry.value_ptr.*.funcs.get(method_name)) |fsig| {
                    return fsig;
                }
            }
        }
    }
}
```

The type-guided path (qualified struct name -> module lookup) is preferred and precise. The fallback iterator only fires when the struct type name is not qualified, which is rare for cross-module types. If the `ResolvedType` struct representation differs from the `.struct_type => |s| s.name` pattern shown above, adapt the field access accordingly — read the `ResolvedType` union definition in `src/types.zig` to get the exact field names. Per D-04 and D-05.
